// 変数に対応
i = 0;

// 条件分岐と論理式
if(i == 0){
    puts("hello");
};

// 四則演算（優先順位を含む）
a = 3 + 4 * 5;  // => 23
puts(a);

// 四則演算（カッコ優先）
b = (3 + 4) * 5;  // => 35
puts(b);

// ユーザーの自作関数定義（階乗を計算する関数）
func fact(n) {
    // ローカル変数
    def i = 1;
    if(n > 0){
        i = n * fact(n - 1);
    };
    i;
};
puts(fact(3)); // 3! => 6
puts(fact(4)); // 4! => 24

// "hello"
// 23
// 35
// 6
// 24
